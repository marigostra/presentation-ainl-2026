\begin{frame}
  \frametitle{Calculation procedure}
  \framesubtitle{Steps for classification }

  
\begin{enumerate}
\item{The~positional encoding (RoPE) is applied to~each token.}
\item{The~dot product attention is calculated inside each word for~~tokens included in~the~word.}
\item{Calculation of~the~score of~the~tokens within the~word.}
\item{Thvectors of~token representation within a~single word are aggregated.}
\item{The~application of positional encoding is repeated once again.}
\item{The~resulting tensor is processed by~four transformer encoder blocks.}
\item{The obtained encoding results are processed by a feed-forward network with ~ one hidden layer.}
\end{enumerate}
  \end{frame}

\begin{frame}
  \frametitle{Model and training}
  \framesubtitle{Technical details of~the~model and its training}

  \begin{block}{Model}
    The~resulting model has the~size of 195MB and has  48792985 weight parameters.
    \end{block}

  \begin{block}{Training}
      The~training process takes approximately 8--12 hours.
The~AdamW optimizer is used with $\lambda = 1e-5$.
The~batch size is equal to 96.
The training process includes 35 epochs with a change in the learning rate in three stages:
for the first 15 epochs, the learning rate was 1e-4;
followed by 10 epochs of learning rate 5e-5;
then 10 more epochs of learning rate 1e-5.
\end{block}
\end{frame}
